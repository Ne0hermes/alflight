(async function() {
  // Importer directement le store
  const storeModule = await import('/src/core/stores/aircraftStore.js');
  const aircraftStore = storeModule.useAircraftStore.getState();

  console.log('Store chargé, nombre d\'avions:', aircraftStore.aircraftList?.length);

  const aircraft = aircraftStore.aircraftList.find(a => a.registration === 'F-HSTR');

  if (!aircraft) {
    console.log('❌ Avion F-HSTR non trouvé');
    console.log('Avions disponibles:', aircraftStore.aircraftList.map(a => a.registration));
    return;
  }

  console.log('✅ Avion F-HSTR trouvé');

  const table = aircraft.advancedPerformance?.tables?.find(t => t.table_name.includes('Landing Distance - Flaps LDG'));

  if (!table) {
    console.log('❌ Tableau Landing non trouvé');
    return;
  }

  console.log('✅ Tableau trouvé:', table.table_name);
  console.log('Points de données:', table.data.length);

  console.log('\n=== MASSE 1100 KG ===');
  const data1100 = table.data
    .filter(p => p.Masse === 1100 && p.Altitude >= 0 && p.Altitude <= 1000 && p.Temperature >= 0 && p.Temperature <= 20)
    .sort((a,b) => a.Altitude - b.Altitude || a.Temperature - b.Temperature);

  data1100.forEach(p => {
    console.log(`${p.Altitude}ft | ${p.Temperature}°C | Roulage: ${p.Distance_roulement}m | 15m: ${p.Distance_passage_15m}m`);
  });

  console.log('\n=== MASSE 1200 KG ===');
  const data1200 = table.data
    .filter(p => p.Masse === 1200 && p.Altitude >= 0 && p.Altitude <= 1000 && p.Temperature >= 0 && p.Temperature <= 20)
    .sort((a,b) => a.Altitude - b.Altitude || a.Temperature - b.Temperature);

  data1200.forEach(p => {
    console.log(`${p.Altitude}ft | ${p.Temperature}°C | Roulage: ${p.Distance_roulement}m | 15m: ${p.Distance_passage_15m}m`);
  });
})();
